77class Segond1910(Bible): 

78 """Louis Segond 1910 French Bible Translation""" 

79 

80 def __init__(self): 

81 super().__init__("Segond 1910") 

82 

83 def load_text(self, file_path: str) -> None: 

84 """Load Segond 1910 bible text from file""" 

85 try: 

86 with open(file_path, 'r', encoding='utf-8') as file: 

87 content = file.read() 

88 # Normalize French chapter markers to English for the parser 

89 content = content.replace("Chapitre", "Chapter") 

90 content = content.replace("Psaume", "Psalm") 

91 self._parse_verse_per_line_format(content) 

92 self._rename_books_to_german() 

93 except Exception as e: 

94 print(f"Error loading Segond 1910 from {file_path}: {e}") 

95 

96 def _rename_books_to_german(self) -> None: 

97 """Rename all French book titles to their German equivalents so that 

98 /api/Segond1910/4%20Mose/1 works just like the German translations.""" 

99 new_books = {} 

100 for fr_name, data in self.books.items(): 

101 german_name = SEGOND1910_TO_GERMAN.get(fr_name, fr_name) 

102 new_books[german_name] = data 

103 self.books = new_books
